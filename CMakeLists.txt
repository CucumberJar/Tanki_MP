cmake_minimum_required(VERSION 3.22)  # 3.30 нет в релизе, лучше 3.22+
project(untitled5)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

set(CMAKE_PREFIX_PATH "D:/QT/6.8.1/mingw_64")

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets Network)
include_directories(
        src
        src/entities
        src/main
)

add_executable(untitled5
        main.cpp
        src/tank/entities/Tank.cpp
        Game.cpp
        Wall.cpp
        Turret.cpp
        Bullet.cpp
        Tile.cpp
        TankClient.cpp
        src/main/GameLoad.cpp
        src/main/GameView.cpp
        src/main/GameScene.cpp

)

target_link_libraries(untitled5 Qt6::Core Qt6::Gui Qt6::Widgets Qt6::Network)

if (WIN32 AND NOT DEFINED CMAKE_TOOLCHAIN_FILE)
    set(DEBUG_SUFFIX "")
    if (MSVC AND CMAKE_BUILD_TYPE MATCHES "Debug")
        set(DEBUG_SUFFIX "d")
    endif ()

    set(QT_INSTALL_PATH "${CMAKE_PREFIX_PATH}")
    if (NOT EXISTS "${QT_INSTALL_PATH}/bin")
        set(QT_INSTALL_PATH "${QT_INSTALL_PATH}/..")
        if (NOT EXISTS "${QT_INSTALL_PATH}/bin")
            set(QT_INSTALL_PATH "${QT_INSTALL_PATH}/..")
        endif ()
    endif ()

    if (EXISTS "${QT_INSTALL_PATH}/plugins/platforms/qwindows${DEBUG_SUFFIX}.dll")
        add_custom_command(TARGET untitled5 POST_BUILD
                COMMAND ${CMAKE_COMMAND} -E make_directory
                "$<TARGET_FILE_DIR:untitled5>/plugins/platforms/")
        add_custom_command(TARGET untitled5 POST_BUILD
                COMMAND ${CMAKE_COMMAND} -E copy
                "${QT_INSTALL_PATH}/plugins/platforms/qwindows${DEBUG_SUFFIX}.dll"
                "$<TARGET_FILE_DIR:untitled5>/plugins/platforms/")
    endif ()

    foreach (QT_LIB Core Gui Widgets)
        add_custom_command(TARGET untitled5 POST_BUILD
                COMMAND ${CMAKE_COMMAND} -E copy
                "${QT_INSTALL_PATH}/bin/Qt6${QT_LIB}${DEBUG_SUFFIX}.dll"
                "$<TARGET_FILE_DIR:untitled5>")
    endforeach (QT_LIB)
endif ()
